# logarithmic regression fitting
def funct(x, p1, p2):
    return p1*np.log(x) + p2

xdata = np.array([x+1 for x in range(len(df))])
ydata = np.log(df['value'])

popt, pcov = curve_fit(funct, xdata, ydata, p0=(3.0, -10))

print(popt)
